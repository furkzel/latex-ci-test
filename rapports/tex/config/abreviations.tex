\chapter*{Liste des abréviations}
\addcontentsline{toc}{chapter}{Liste des abréviations}

\begin{description}[leftmargin=4cm, labelwidth=3.5cm, style=nextline]

    \item[CVaR] Conditional Value-at-Risk (Valeur à Risque Conditionnelle)

    \item[VaR] Value-at-Risk (Valeur à Risque)

    \item[ES] Expected Shortfall (Perte Espérée Conditionnelle)

    \item[DRO] Distributionally Robust Optimization (Optimisation Robuste par Rapport à la Distribution)

    \item[PL] Programmation Linéaire (Linear Programming)

    \item[MC] Monte Carlo

\end{description}
